generator client {
  provider = "prisma-client-js"
  output   = "../src/lib/generated/prisma"

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

/* =========================
   ENUMS
   ========================= */

enum UserRole {
  ADMIN
  USER

  @@map("user_role")
}

enum MerchantStatus {
  ACTIVE
  INACTIVE

  @@map("merchant_status")
}

enum ApiKeyStatus {
  ACTIVE
  REVOKED

  @@map("api_key_status")
}

enum ProviderName {
  TRANSFEERA
  WOOVI

  @@map("provider_name")
}

enum ChargeStatus {
  CREATING
  PENDING
  PAID
  CANCELED
  EXPIRED
  FAILED

  @@map("charge_status")
}

enum WithdrawalStatus {
  REQUESTED
  PROCESSING
  PAID
  FAILED
  CANCELED

  @@map("withdrawal_status")
}

enum LedgerEntryType {
  CREDIT
  DEBIT
  FEE

  @@map("ledger_entry_type")
}

enum LedgerReferenceType {
  CHARGE
  WITHDRAWAL
  ADJUSTMENT

  @@map("ledger_reference_type")
}

/* =========================
   MODELS
   ========================= */

model User {
  id           String   @id @default(uuid()) @map("id")
  email        String   @unique @map("email")
  passwordHash String   @map("password_hash")
  role         UserRole @default(USER) @map("role")

  createdAt DateTime @default(now()) @map("created_at")
  updatedAt DateTime @updatedAt @map("updated_at")

  // MVP simples: 1 user -> 1 merchant
  merchant Merchant?

  @@map("users")
  @@index([role], map: "users_role_idx")
}

model Merchant {
  id String @id @default(uuid()) @map("id")

  userId String @unique @map("user_id")
  user   User   @relation(fields: [userId], references: [id], onDelete: Cascade)

  name  String @map("name")
  email String @map("email")
  phone String @map("phone")

  status MerchantStatus @default(ACTIVE) @map("status")

  // provider ativo (troca por adapter sem mudar API)
  provider ProviderName @default(TRANSFEERA) @map("provider")

  createdAt DateTime @default(now()) @map("created_at")
  updatedAt DateTime @updatedAt @map("updated_at")

  apiKeys       ApiKey[]
  charges       Charge[]
  withdrawals   Withdrawal[]
  ledgerEntries LedgerEntry[]

  @@map("merchants")
  @@index([status], map: "merchants_status_idx")
  @@index([provider], map: "merchants_provider_idx")
}

model ApiKey {
  id         String       @id @default(uuid()) @map("id")
  merchantId String       @map("merchant_id")

  name    String @map("name")
  prefix  String @map("prefix")
  last4   String @map("last4")
  keyHash String @unique @map("key_hash")

  status    ApiKeyStatus @default(ACTIVE) @map("status")
  revokedAt DateTime?    @map("revoked_at")

  createdAt DateTime @default(now()) @map("created_at")

  merchant Merchant @relation(fields: [merchantId], references: [id], onDelete: Cascade)

  @@map("api_keys")
  @@index([merchantId, status], map: "api_keys_merchant_status_idx")
  @@index([prefix], map: "api_keys_prefix_idx")
}

model Charge {
  id         String @id @default(uuid()) @map("id")
  merchantId String @map("merchant_id")

  amount      Int     @map("amount") // cents
  description String? @map("description")
  status      ChargeStatus @default(CREATING) @map("status")

  provider         ProviderName @default(TRANSFEERA) @map("provider")
  providerChargeId String?      @map("provider_charge_id")

  idempotencyKey String @map("idempotency_key")

  pixQrCode    String?   @map("pix_qr_code")
  pixCopyPaste String?   @map("pix_copy_paste")
  expiresAt    DateTime? @map("expires_at")
  paidAt       DateTime? @map("paid_at")

  metadata Json? @map("metadata")

  createdAt DateTime @default(now()) @map("created_at")
  updatedAt DateTime @updatedAt @map("updated_at")

  merchant Merchant @relation(fields: [merchantId], references: [id], onDelete: Cascade)

  @@map("charges")
  @@unique([merchantId, idempotencyKey], map: "charges_merchant_idempotency_key_uniq")
  @@unique([provider, providerChargeId], map: "charges_provider_provider_charge_id_uniq")
  @@index([merchantId, createdAt], map: "charges_merchant_created_at_idx")
  @@index([status], map: "charges_status_idx")
}

model Withdrawal {
  id         String @id @default(uuid()) @map("id")
  merchantId String @map("merchant_id")

  amount      Int     @map("amount") // cents
  description String? @map("description")
  status      WithdrawalStatus @default(REQUESTED) @map("status")

  provider             ProviderName @default(TRANSFEERA) @map("provider")
  providerWithdrawalId String?      @map("provider_withdrawal_id")

  pixKey     String @map("pix_key")
  pixKeyType String @map("pix_key_type")

  idempotencyKey String? @map("idempotency_key")

  createdAt DateTime @default(now()) @map("created_at")
  updatedAt DateTime @updatedAt @map("updated_at")

  merchant Merchant @relation(fields: [merchantId], references: [id], onDelete: Cascade)

  @@map("withdrawals")
  @@unique([provider, providerWithdrawalId], map: "withdrawals_provider_provider_withdrawal_id_uniq")
  @@unique([merchantId, idempotencyKey], map: "withdrawals_merchant_idempotency_key_uniq")
  @@index([merchantId, createdAt], map: "withdrawals_merchant_created_at_idx")
  @@index([status], map: "withdrawals_status_idx")
}

// Ledger imutável: nunca update/delete no app (só insert)
model LedgerEntry {
  id         String @id @default(uuid()) @map("id")
  merchantId String @map("merchant_id")

  entryType LedgerEntryType @map("entry_type")
  amount    Int             @map("amount") // cents

  referenceType LedgerReferenceType @map("reference_type")
  referenceId   String              @map("reference_id")

  description String? @map("description")

  createdAt DateTime @default(now()) @map("created_at")

  merchant Merchant @relation(fields: [merchantId], references: [id], onDelete: Cascade)

  @@map("ledger_entries")
  @@index([merchantId, createdAt], map: "ledger_entries_merchant_created_at_idx")
  @@index([referenceType, referenceId], map: "ledger_entries_reference_idx")
  @@unique([merchantId, entryType, referenceType, referenceId], map: "ledger_entries_idempotency_uniq")
}
